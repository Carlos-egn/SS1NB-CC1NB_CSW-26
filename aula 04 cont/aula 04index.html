        import math

    print("=== Calculadora de Equação do 2º Grau ===")

    # Entrada de dados
    a = float(input("Digite o valor de a: "))
    b = float(input("Digite o valor de b: "))
    c = float(input("Digite o valor de c: "))

    # Verificação
    if a == 0:
    print("Não é uma equação do 2º grau (a precisa ser diferente de 0).")
    else:
    delta = b**2 - 4*a*c
    print(f"\nDelta = {delta}")

    if delta > 0:
    x1 = (-b + math.sqrt(delta)) / (2*a)
    x2 = (-b - math.sqrt(delta)) / (2*a)
    print("Duas raízes reais diferentes:")
    print(f"x1 = {x1}")
    print(f"x2 = {x2}")

    elif delta == 0:
    x = -b / (2*a)
    print("Uma raiz real (raiz dupla):")
    print(f"x = {x}")

    else:
    real = -b / (2*a)
    imag = math.sqrt(-delta) / (2*a)
    print("Duas raízes complexas:")
    print(f"x1 = {real} + {imag}i")
    print(f"x2 = {real} - {imag}i")
